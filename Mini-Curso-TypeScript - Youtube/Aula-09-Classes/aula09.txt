private // Significa que nenhum lugar externo vai conseguir chamar e nem alterar esta propriedade.

readonly // Significa que o atributo pode ser lido porem não editado.

public // Significa que ela é publica não existe nenhuma regra.

protected // Significa que a gente consegue chamar dentro da Classe ou das suas subClasses somente.


// Exemplo do metodo GET
get getLevel() {
   console.log("----GET----");
   return this.level;
}

Para chamar ele: 
> console.log(john.getLevel);


// Exemplo do metodo SET
set setLevel(level: number) {
   this.level = level;
}

Para usar ele: 
> john.setLevel = 499;
console.log(john.getLevel);

// Abstract Class

O que é?
R: Basicamente é uma classe abstrata, onde não é possivel criar objetos a partir dela, porem eu posso extender ela. A classe abstrata é interessante quando a 
gente quer criar classes que irão ser extendida.
